    import numpy as np
from multiprocessing import cpu_count

from .base import ModelSecondOrder, ModelSelfConcordant, LOSS_AND_GRAD

    _ModelHawkesFixedExpKernLogLik



    [docs]class ModelHawkesFixedExpKernLogLik(ModelSecondOrder,
                                    ModelSelfConcordant):
    """
    Model for Hawkes processes with fixed exponential kernels using the
    log-likelihood contrast.
    For now, all kernels are equal to an exponential and a fixed decay
    across nodes.

    Parameters
    ----------
    decays : `float`
        The decay coefficient of the exponential kernel

    n_threads : `int`, default=-1 (read-only)
        Number of threads used for parallel computation.

        * if ``int <= 0``: the number of physical cores available on
          the CPU
        * otherwise the desired number of threads

    Attributes
    ----------
    n_nodes : `int` (read-only)
        Number of components, or dimension of the Hawkes model

    data : `list` of `numpy.array` (read-only)
        The events of each component of the Hawkes. Namely
        ``data[j]`` contains a one-dimensional `numpy.array` of
        the events' timestamps of component j
    """
    # In Hawkes case, getting value and grad at the same time need only
    # one pas over the data
    pass_per_operation = \
        {k: v for d in [ModelSecondOrder.pass_per_operation,
                        {LOSS_AND_GRAD: 1}] for k, v in d.items()}

    _attrinfos = {
        "decays": {
            "cpp_setter": "set_decay"
        },
        "dim": {
            "writable": False
        },
        "n_threads": {
            "cpp_setter": "set_n_threads"
        },
        "data": {
            "writable": False
        },
    }

    def __init__(self, decays: np.ndarray, n_threads: int = -1):
        ModelSecondOrder.__init__(self)
        ModelSelfConcordant.__init__(self)
        if n_threads <= 0:
            n_threads = cpu_count()
        self.decays = decays
        self.n_threads = n_threads
        self._model = None
        self.dim = None
        self.data = None

    def _get_n_coeffs(self):
        return self._model.get_n_coeffs()


    [docs]    def fit(self, data):
        """Set the data into the model object

        Parameters
        ----------
        data : `list` of `numpy.array`
            The events of each component of the Hawkes. Namely
            ``data[j]`` contains a one-dimensional `numpy.array` of
            the events' timestamps of component j
        """
        ModelSecondOrder.fit(self, data)
        ModelSelfConcordant.fit(self, data)
        self._set("_model", _ModelHawkesFixedExpKernLogLik(self.decays,
                                                           data,
                                                           self.n_threads))
        return self
    

    def _set_data(self, data):
        dim = len(data)
        self._set("dim", dim)
        self._set("data", data)

    def _loss(self, coeffs: np.ndarray) -> float:
        return self._model.loss(coeffs)

    def _grad(self, coeffs: np.ndarray, out: np.ndarray) -> np.ndarray:
        self._model.grad(coeffs, out)

    def _loss_and_grad(self, coeffs: np.ndarray, out: np.ndarray):
        value = self._model.loss_and_grad(coeffs, out)
        return value

    def _hessian_norm(self, coeffs: np.ndarray,
                      point: np.ndarray) -> float:
        return self._model.hessian_norm(coeffs, point)

    def _get_sc_constant(self) -> float:
        return 2.0

    @property
    def n_jumps(self):
        return sum(map(len, self.data))

    @property
    def _epoch_size(self):
        # This gives the typical size of an epoch when using a
        # stochastic optimization algorithm
        return self.n_jumps

    @property
    def _rand_max(self):
        # This allows to obtain the range of the random sampling when
        # using a stochastic optimization algorithm
        return self.n_jumps
